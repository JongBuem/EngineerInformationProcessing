Q문제 코드
C 
#include <stdio.h>

typedef struct {
    int    accNum;
    double bal;
} BankAcc;

double sim_pow(double base, int year) {
    double r = 1.0;
    for (int i = 0; i < year; i++) r = r * base;
    return r;
}

void initAcc(BankAcc *acc, int x, double y) {
    acc->accNum = x;
    acc->bal   = y;
}

void xxx(BankAcc *acc, double *en) {  // 출금 or 입금
    if (*en > 0 && *en < acc->bal)
        acc->bal = acc->bal - *en;  // 출금
    else
        acc->bal = acc->bal + *en;  // 입금
}

void yyy(BankAcc *acc) {              // 3년 10% 복리
    acc->bal = acc->bal * sim_pow((1+0.1), 3);
}

int main() {
    BankAcc myAcc;
    initAcc(&myAcc, 9981, 2200.0);
    double amount = 100.0;
    xxx(&myAcc, &amount);
    yyy(&myAcc);
    printf("%d and %.2f", myAcc.accNum, myAcc.bal);
    return 0;
}

① 문제 유형

구조체 포인터를 여러 함수에 전달하며 필드 값이 단계적으로 변경되는 상태 추적 문제입니다.


  • 🏦initAcc: 초기화 / xxx: 출금 or 입금 조건 분기 / yyy: 3년 복리 계산의 3단계 함수 체인입니다.
  • ⚠️함정 — xxx 조건 분기: *en > 0 && *en < acc->bal 두 조건을 모두 확인해야 출금/입금 여부가 결정됩니다.
  • 🔢sim_pow(1.1, 3)은 1.1³ = 1.331을 반환합니다.

⑦ 핵심 주의 포인트
  • 🏦 xxx 분기 조건: en > 0 && en < bal이면 출금(빼기), 그 외면 입금(더하기). amount=100은 두 조건을 모두 통과 → 출금 실행.
  • 📈 sim_pow(1.1, 3): 직접 반복문으로 1.1을 3번 곱해 1.331을 반환합니다. 10% 복리 3년 이율 계수입니다.
  • 🔢 %.2f: 소수점 둘째 자리까지 출력 → 2795.1 → 2795.10
  • 🖨️ 최종 출력: 9981 and 2795.10
